function grade(marks){
    if(marks >=100){
        return "Ente Marks Between 0-100";
    }
    else if(marks >=90){
        return "You Got Grade A";
    }
    else if(marks >=80){
        return "You Got Grade B";
    }
    else if(marks >=70){
        return "You Got Grade C";
    }
    else if(marks >=60){
        return "You Got Grade D";
    }
    else if(marks <= 59){
        return "You Got Grade E";
    }
    else {
        return "You Enter Invalid Number";
    }
}
